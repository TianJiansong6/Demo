int main ()
{	int num,ge,shi,bai,res;
	printf ("输入一个三位数\n");
	scanf("%d", &num);
	if(num>=100&&num<=999){		//判断是否合要求
		ge=num % 10;		//取出各个位
		shi= num/10%10; 
		bai= num/100;  
		res = bai+shi*10+ge*100;
		printf("逆序后的数为%d\n", res);	//result
		}
	else
		printf("illegal input");
}

	for (n=1;n<=a&&n<=b; n++){	//最大公约数比两个数都小

			t=n;
		}

#include<stdio.h>
int main ()
{	
	printf ("输入两个数a和b:\n");
	int a,b,t;
	scanf("%d %d",&a,&b);
	if(a>b){			//a?b 否则会报错
		t=a%b;
		if(t==0){
			printf("a,b的最大公约数为%d\n",b);}	//先判断b是否为a的直接最大公约数
		else{
	
			for(;t!=0;)				//for循环，辗转相除法
			{
			t=a%b;
			a = b;
			b = t;					
			}
			printf("a,b的最大公约数为%d\n",a);
			}
			}
	else
		printf("illegal input");
}
