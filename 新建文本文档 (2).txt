#include"stdio.h"
void main(){
int i=0;
printf("请输入两个数（大数在前）");
int a,b,c,d;
scanf("%d%d",&a,&b);
if(a%2==0&&b%2==0)
d=0;
else
d=1;
switch(d)
{
case 0:
		do
		{
		i++;
		a=a/2;
		b=b/2;
		}
		while(a%2==0&&b%2==0);
case 1:
{
		do
		{
			c=a-b;
			if(c<b)
					{
						a=b;
						b=c;
					}
			else
				a=c;
		}
		while(b!=c);
        for(;i<=0;i--)
		c=c*2;
}
printf("%d",c);
}
}
